📤 Contoh Request
javascript
// Kawaiiyumee File API — JavaScript (fetch)
const uploadToKawaiiyumee = async (file) => {
  const formData = new FormData()
  formData.append("file", file)

  const res = await fetch(
    "https://api.kawaiiyumee.web.id/api/tools/tourl",
    { method: "POST", body: formData }
  )
  const data = await res.json()

  if (data.status) {
    console.log("✅ URL:", data.url)
  } else {
    console.error("❌ Gagal:", data.message)
  }
}

# Kawaiiyumee File API — Python
import requests

def upload(path):
    with open(path, "rb") as f:
        res = requests.post(
            "https://api.kawaiiyumee.web.id/api/tools/tourl",
            files={"file": (path.split("/")[-1], f)}
        )
    data = res.json()
    print(data["url"] if data["status"] else data["message"])

<?php
function uploadToKawaiiyumee($path) {
    $ch = curl_init("https://api.kawaiiyumee.web.id/api/tools/tourl");
    curl_setopt_array($ch, [
        CURLOPT_POST           => true,
        CURLOPT_POSTFIELDS     => ["file" => new CURLFile(realpath($path))],
        CURLOPT_RETURNTRANSFER => true,
    ]);
    $data = json_decode(curl_exec($ch), true);
    curl_close($ch);
    echo $data["status"] ? $data["url"] : $data["message"];
}
