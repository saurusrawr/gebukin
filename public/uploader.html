Dokumentasi API
POST api.kawaiiyumee.web.id/api/tools/tourl

Endpoint ini menerima file melalui multipart/form-data dan mengembalikan URL File permanen yang bisa langsung dipakai. File disimpan secara permanen — tidak ada batas waktu, tidak ada penghapusan! 🌸

Field Tipe Wajib Keterangan
file File (multipart) WAJIB File yang ingin diupload. Gunakan multipart/form-data dengan field name file

📤 Contoh Request
javascript
// Kawaiiyumee File API — JavaScript (fetch)
const uploadToKawaiiyumee = async (file) => {
  const formData = new FormData()
  formData.append("file", file)

  const res = await fetch(
    "https://api.kawaiiyumee.web.id/api/tools/tourl",
    { method: "POST", body: formData }
  )

  const data = await res.json()

  if (data.status) {
    console.log("✅ Berhasil! URL:", data.url)
    // data.url  → URL File permanen
    // data.filename → nama file (misal: a1b2c3.jpg)
    // data.size → ukuran file dalam byte
    // data.type → MIME type file
  } else {
    console.error("❌ Gagal:", data.message)
  }
}

// Contoh penggunaan dengan file input HTML:
document.querySelector("#fileInput").addEventListener("change", (e) => {
  uploadToKawaiiyumee(e.target.files[0])
})

# Kawaiiyumee File API — Python (requests)
import requests

def upload_to_kawaiiyumee(file_path):
    url = "https://api.kawaiiyumee.web.id/api/tools/tourl"

    with open(file_path, "rb") as f:
        files = {
            "file": (
                file_path.split("/")[-1],  # nama file
                f,                           # file object
            )
        }
        res = requests.post(url, files=files)
        data = res.json()

    if data["status"]:
        print(f"✅ Berhasil! URL: {data['url']}")
        return data["url"]
    else:
        print(f"❌ Gagal: {data['message']}")
        return None

# Contoh penggunaan:
upload_to_kawaiiyumee("./foto.jpg")
<?php
// Kawaiiyumee File API — PHP (cURL)

function uploadToKawaiiyumee($filePath) {
    $url = "https://api.kawaiiyumee.web.id/api/tools/tourl";

    $cfile = new CURLFile(
        realpath($filePath),
        mime_content_type($filePath),
        basename($filePath)
    );

    $ch = curl_init($url);
    curl_setopt_array($ch, [
        CURLOPT_POST           => true,
        CURLOPT_POSTFIELDS     => ["file" => $cfile],
        CURLOPT_RETURNTRANSFER => true,
    ]);

    $response = curl_exec($ch);
    curl_close($ch);

    $data = json_decode($response, true);

    if ($data["status"]) {
        echo "✅ URL: " . $data["url"] . "\n";
        return $data["url"];
    }

    echo "❌ Error: " . $data["message"] . "\n";
    return null;
}

// Contoh penggunaan:
uploadToKawaiiyumee("./foto.jpg");
?>
📥 Response Schema

API mengembalikan JSON dengan struktur berikut:

// ✅ Response Berhasil (HTTP 200)
{
  "status":   true,                                      // boolean — selalu true jika sukses
  "filename": "a1b2c3.jpg",                              // nama file unik di File
  "url":      "https://cdn.kawaiiyumee.web.id/a1b2c3.jpg", // URL File permanen ✨
  "size":     204800,                                    // ukuran file dalam byte
  "type":     "image/jpeg"                               // MIME type file
}

// ❌ Response Gagal (HTTP 400 / 500)
{
  "status":  false,
  "message": "Pesan error di sini"
}

// Kemungkinan error:
// "File wajib diupload. Gunakan form-data dengan field 'file'"
// "Tipe file '.exe' tidak diizinkan"
// "File terlalu besar. Maksimum 25 MB"
